CREATE TYPE tournament_category AS enum('Junior', 'Senior', 'Veteran');
CREATE TYPE tournament_status AS enum('Waiting for players', 'In progress', 'Completed');
create table tournament (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),

    name varchar(63) not null,
    location varchar(120) not null,
    min_player_allow smallint default 2 not null check(min_player_allow between 2 and 32),
    max_player_allow smallint not null check(max_player_allow between 2 and 32),
    min_elo_allow smallint default 0 not null,  -- 1200 min.
    max_elo_allow smallint not null,
    category tournament_category not null,  -- can allow many different ? (use flag).
    status tournament_status default 'Waiting for players' not null,
    current_round smallint default 0 not null,
    is_women_only bool default False not null,
    date date not null,

    --constraint pk primary key(id)  -- already set in id declaration.
    constraint ck_players check(min_player_allow <= max_player_allow)
);

CREATE TYPE registration_status AS enum('registered', 'cancel');
create table registration (
    id_player uuid not null references player(id) on delete cascade,
    id_tournament uuid not null references tournament(id) on delete cascade,

    registration_date timestamptz default now() not null,
    status registration_status default 'registered' not null,

    constraint pk primary key (id_player, id_tournament)
